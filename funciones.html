b=30
c=60

def f(x,a,b,c):
    
   
    if (x<a):
        p=0
  
    if (x>=a)&(x<b):
        p=2*(((x-a)/(c-a))**2)
    
    if (x>=b)&(x<=c):
        p=1-2*(((x-c)/(c-a))**2)
   
    if (x>c):
        p=1

    return p
